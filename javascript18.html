Switch case in JavaScript

Sample Program :

<!Doctype Html >
<html >
<body>
<script type="text/javascript">
var king = "ramesh";
switch(king)
{
case "varma":
document.write("varma is the best ruler");
break;
case "ramesh":
document.write("ramesh is the best teacher");
break;
case "madhav":
document.write("madhav is the best student");
break;
case "pavan":
document.write("pavan is the good brother");
break;
}
</script>
</body>
</html>
